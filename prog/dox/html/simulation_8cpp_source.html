213  void compute_accel( params_t* params)
214  {
215  // Unpack basic parameters
216  const float h = params->h;
217  const float rho0 = params->rho0;
218  const float h2 = h*h;
219  // Compute density and color
220  compute_density(params);
221  // Start with gravity and surface forces
222  Vector gravity(params->gx, params->gy);
223  for (unsigned i = 0; i < n; ++i) {
224  a[i] = gravity;
225  }
226  // Constants for interaction term
227  float C0 = params->mass / M_PI / ( h2*h2 );
228  float Cp = 15 * params->k;
229  float Cv = -40 * params->mu;
230  // Now compute interaction forces
231  for (unsigned i = 0; i < n; ++i) {
232  const float rhoi = rho[i];
233  for (unsigned j = i+1; j < n; ++j) {
234  Vector dp = p[i] - p[j];
235  float r2 = dp.normSquared();
236  if (r2 < h2) {
237  const float rhoj = rho[j];
238  Vector dv = v[i] - v[j];
239  float q = sqrt(r2) / h;
240  float u = 1-q;
241  float w0 = C0 * u/rhoi/rhoj;
242  float wp = w0 * Cp * (rhoi+rhoj-2*rho0) * u/q;
243  float wv = w0 * Cv;
244 
245  a[i] += dp*wp + dv*wv;
246  a[j] -= dp*wp + dv*wv;
247  }
248  }
249  }
