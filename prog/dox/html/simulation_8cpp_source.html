62 {
63  // Unpack basic parameters
64  const float h = params->h;
65  const float rho0 = params->rho0;
66  const float h2 = h*h;
67  // Compute density and color
68  computeDensity(params);
69  // Start with gravity and surface forces
70  Vector gravity(params->gx, params->gy);
71  for (unsigned i = 0; i < n; ++i) {
72  a[i] = gravity;
73  }
74  // Constants for interaction term
75  float C0 = params->mass / M_PI / ( h2*h2 );
76  float Cp = 15 * params->k;
77  float Cv = -40 * params->mu;
78  // Now compute interaction forces
79  for (unsigned i = 0; i < n; ++i) {
80  const float rhoi = rho[i];
81  for (unsigned j = i+1; j < n; ++j) {
82  Vector dp = p[i] - p[j];
83  float r2 = dp.normSquared();
84  if (r2 < h2) {
85  const float rhoj = rho[j];
86  Vector dv = v[i] - v[j];
87  float q = sqrt(r2) / h;
88  float u = 1-q;
89  float w0 = C0 * u/rhoi/rhoj;
90  float wp = w0 * Cp * (rhoi+rhoj-2*rho0) * u/q;
91  float wv = w0 * Cv;
92 
93  a[i] += dp*wp + dv*wv;
94  a[j] -= dp*wp + dv*wv;
95  }
